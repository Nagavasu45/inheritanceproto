Q1. Create 2 objects - parent and child. Attach methods to parent and use those methods in child object using parents prototype 

    function Parent(name) {
        this.name=name;
    }
    Parent.prototype.first = function() {
        console.log(this.name)
    }

    function Child(name,greet){
        Parent.call(this, name);
        this.greet =greet;
    }

    Child.prototype = Object.create(Parent.prototype);

    Child.prototype.second = function() {
                console.log(this.name,this.greet);
    };

    var parent = new Parent('Vasu');
    var child = new Child('Vasuvasu', " hi");

    parent.first(); 
    child.first();
    child.second()      
    


    output:    
    Vasu
    Vasuvasu
    Vasuvasu  hi

Q2. Write code to explain prototype chaining 

        function Person() {
            this.name= 'Vasu'
        }
        Person.prototype.name = 'Naga';
        Person.prototype.age = 9;
        const person1 = new Person();
        console.log(person1.name); 
        console.log(person1.age); 

    

output:Vasu
        9

explaination: In the above code, there is a function like Person() and inside we have this.name='Vasu' and then Person.prototype.name='Naga' and one more is age propertie and then one object is created for the function and call the name and age with using created object,and here the concept is if you want to get the name value then it will search in inside of function and then if there is no any name variable and then go to inherited names or properties if it find anything with name variable we get the output this kind of process is called prototype chaining

Q4. Write a JavaScript function to retrieve all the names of object's own and inherited properties. 

function allKeys(obj) {
        if (!isObject(obj)) return [];
        var keys = [];
        for (var key in obj) keys.push(key);
        return keys;
    }
    function isObject(obj) 
    {
        var type = typeof obj;
        return type === 'function' || type === 'object' && !!obj;
      }
    function Student(name) {
      this.name = name;
    }
    Student.prototype.rollno = 9;
    console.log(allKeys(new Student("vasu")));
    
            

    

output:[name,rollno]
